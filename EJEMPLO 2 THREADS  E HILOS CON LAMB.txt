EJEMPLO 2 THREADS  E HILOS CON LAMBDA
public class EjemploHilosLambda {
    public static void main(String[] args) {
        // Crear un objeto Runnable usando una expresi贸n lambda
        Runnable tarea1 = () -> {
            for (int i = 1; i <= 5; i++) {
                System.out.println("Tarea 1 - Iteraci贸n " + i);
                try {
                    Thread.sleep(1000);
                } catch (InterruptedException e) {
                    e.printStackTrace();
                }
            }
        };

        // Crear un objeto Runnable usando una clase an贸nima
        Runnable tarea2 = new Runnable() {
            @Override
            public void run() {
                for (int i = 1; i <= 5; i++) {
                    System.out.println("Tarea 2 - Iteraci贸n " + i);
                    try {
                        Thread.sleep(1000);
                    } catch (InterruptedException e) {
                        e.printStackTrace();
                    }
                }
            }
        };

        // Crear dos hilos y ejecutar las tareas en paralelo
        Thread hilo1 = new Thread(tarea1);
        Thread hilo2 = new Thread(tarea2);
        hilo1.start();
        hilo2.start();

        // Esperar a que los hilos terminen antes de finalizar el programa
        try {
            hilo1.join();
            hilo2.join();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        System.out.println("Programa finalizado.");
    }
}
